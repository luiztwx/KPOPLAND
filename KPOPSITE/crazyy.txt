<script>
    const timings = [
        0,   // Line 1
        3,   // Line 2
        6,   // Line 3
        9,   // Line 4
        12,  // Line 5
        15,  // Line 6
        18,  // Line 7
        21,  // Line 8
        24,  // Line 9
        27,  // Line 10
        // Adicione os tempos correspondentes para as demais linhas
    ];

    let currentLine = 0;
    const audio = document.getElementById("music");

    audio.ontimeupdate = () => {
        if (currentLine < timings.length && audio.currentTime >= timings[currentLine]) {
            document.querySelectorAll('.lyrics p').forEach(p => p.style.display = 'none'); // Esconde todas as linhas
            document.getElementById(`line${currentLine + 1}`).style.display = 'block'; // Mostra a linha atual
            currentLine++;
        }
    };

    function openPopup(title, lyrics) {
        document.getElementById("popupTitle").innerText = title;
        document.getElementById("popupLyrics").innerHTML = lyrics;
        document.getElementById("lyricsPopup").style.display = "block";
        audio.play(); // Inicia a música ao abrir o popup
        currentLine = 0; // Reseta a linha atual
        document.querySelectorAll('.lyrics p').forEach(p => p.style.display = 'none'); // Esconde todas as linhas
    }

    function closePopup() {
        document.getElementById("lyricsPopup").style.display = "none";
        audio.pause(); // Para a música ao fechar o popup
    }

    window.onclick = function(event) {
        const popup = document.getElementById("lyricsPopup");
        if (event.target === popup) {
            closePopup();
        }
    }
</script>
